                #include <iostream>
                #include <vector>
                
                using namespace std;
                
                template<class T>
                vector<T> Merge(vector<T> a, vector<T> b)
                {
                    vector<T> c;
                
                    int i = 0, 
                        j = 0;
                
                    while (true)
                    {
                        if (i >= a.size() || j >= b.size()) 
                            break;
                        if (a[i] < b[j])
                        {
                            c.push_back(a[i]);
                            i++;
                        } 
                        else 
                        {
                            c.push_back(b[j]);
                            j++;
                        }
                    }
                    while (i < a.size()) c.push_back(a[i++]);
                    while (j < b.size()) c.push_back(b[j++]);
                    return c;
                } 
                
                template<class T>
                vector<T> Sort(vector<T> a)
                {
                    if (a.size() < 2) return a;
                    return Merge(Sort(
                                       vector<T>(
                                                  a.begin(), 
                                                  a.begin() + a.size() / 2
                                                )
                                      ), 
                                 Sort(
                                       vector<T>(
                                                  a.begin() + a.size() / 2, 
                                                  a.end()
                                                )
                                     )
                                );
                }
                
                int main()
                {
                    int n;
                    cin >> n;
                    vector<int> a(n);
                    for (int i = 0; i < n; i++)
                        cin >> a[i];
                    a = Sort(a);
                    for (int i = 0; i < n; i++)
                        cout << a[i] << ' ';
                }
