                            using namespace std;
                            
                            template
                            vector Merge(vector a, vector b)
                            {
                                vector c;
                                
                                int i = 0, 
                                    j = 0;
                                
                                while (true)
                                {
                                    if (i >= a.size() || j >= b.size()) 
                                        break;
                                    if (a[i] < b[j])
                                    {
                                        c.push_back(a[i]);
                                        i++;
                                    } 
                                    else 
                                    {
                                        c.push_back(b[j]);
                                        j++;
                                    }
                                }
                                while (i < a.size()) c.push_back(a[i++]);
                                while (j < b.size()) c.push_back(b[j++]);
                                return c;
                            } 
                            
                            template
                            vector Sort(vector a)
                            {
                                if (a.size() < 2) return a;
                                return Merge(Sort(
                                                   vector(
                                                              a.begin(), 
                                                              a.begin() + a.size() / 2
                                                            )
                                                  ), 
                                             Sort(
                                                    vector(
                                                               a.begin() + a.size() / 2, 
                                                               a.end()
                                                             )
                                                  )
                                            );
                            }
                            
                            int main()
                            {
                                int n;
                                cin >> n;
                                vector a(n);
                                for (int i = 0; i < n; i++)
                                    cin >> a[i];
                                a = Sort(a);
